Let's make a new module


Here is an example module script 

# module name
def mname
  "example"
end

# module commands, generate pre-defined
def cmds
  [ "generate", "example" ]
end

# custom variables to be set, used by the tab completer
# retrievable via vhash passed to generate function
def vars
  [ "example", "file", "whatever" ]
end

# use built-in extensions
def use
  [ "msfvenom" ]
end

# Defined generation function
def generate(vhash)
    # vhash: a hash which contains variables and their values
    # do what you want here
end

# Anything that isn't a class, module, or function is ignored
